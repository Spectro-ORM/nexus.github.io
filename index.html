Core

Plug Is Just a Function

No protocols to conform to. Plug is a typealias for @Sendable (Connection) async throws -> Connection. Write a closure. Get middleware.

let auth: Plug = { conn in
    guard let token = conn.request
        .headerFields[.authorization] else {
        return conn.respond(
            status: .unauthorized,
            body: .string("No token")
        )
    }
    return conn.assign(key: "token", value: token)
}
Composition

pipe & pipeline

Compose any two plugs with pipe(a, b). Chain a list with pipeline([…]). Short-circuits on halt.

Value Types

Connection Is a Struct

No shared mutable state. Every plug returns a new Connection. Thread-safe by construction.

Routing

Result-Builder DSL

Router { GET("/health") { … } } — declare routes with Swift's native builder syntax. Supports if, for, scopes.

Concurrency

Swift 6 From Day One

Strict concurrency. Everything Sendable. No @unchecked escape hatches in the core target.

Adapters

Server-Agnostic

Router

Scoped Routes with Middleware

Group routes under a prefix with scope("/api"). Pipe middleware through scopes with through:. Scopes nest — prefixes and middleware compound.

let router = Router {
    GET("/health") { conn in
        conn.respond(status: .ok, body: .string("OK"))
    }
    scope("/api", through: [auth]) {
        GET("/users") { conn in
            conn.respond(status: .ok, body: .string("[]"))
        }
        GET("/users/:id") { conn in
            let id = conn.params["id"] ?? ""
            return conn.respond(
                status: .ok, body: .string(id)
            )
        }
    }
}

Composable Plugs

A Plug is just a function. Compose with pipe or pipeline. Each plug returns a new Connection — the original is never mutated.

Pipeline.swift
// A plug is a function: (Connection) async throws -> Connection
let logger: Plug = { conn in
    print("→ \(conn.request.method) \(conn.request.path ?? "/")")
    return conn
}

let jsonHeader: Plug = { conn in
    var copy = conn
    copy.response.headerFields[.contentType] = "application/json"
    return copy
}

// Compose into a pipeline — halts short-circuit
let app = pipeline([logger, jsonHeader, router])

Router DSL

Declare routes with Swift's result-builder syntax. Supports GET, POST, PUT, DELETE, PATCH, HEAD, OPTIONS. First match wins.

Router.swift
let router = Router {
    GET("/health") { conn in
        conn.respond(status: .ok, body: .string("OK"))
    }

    POST("/users") { conn in
        conn.respond(status: .created, body: .string("created"))
    }

    DELETE("/users/:id") { conn in
        conn.respond(status: .noContent, body: .empty)
    }
}

// Use the router as a plug
let app = pipe(logger, router)

Path Parameters

Use :name segments in your path pattern. Captured values land in conn.params. Percent-decoded automatically.

Params.swift
let router = Router {
    // Single parameter
    GET("/users/:id") { conn in
        let id = conn.params["id"] ?? ""
        return conn.respond(
            status: .ok,
            body: .string("User \(id)")
        )
    }

    // Multiple parameters
    GET("/users/:userId/posts/:postId") { conn in
        let userId = conn.params["userId"] ?? ""
        let postId = conn.params["postId"] ?? ""
        return conn.respond(
            status: .ok,
            body: .string("\(userId)/\(postId)")
        )
    }
}

Scoped Routes & Middleware

Group routes under a prefix. Pipe middleware through scopes with through:. Scopes nest — outer middleware wraps inner.

Scopes.swift
let auth: Plug = { conn in /* verify token */ }
let rateLimit: Plug = { conn in /* check rate */ }

let router = Router {
    GET("/health") { conn in
        conn.respond(status: .ok, body: .string("OK"))
    }

    scope("/api", through: [auth]) {
        GET("/users") { conn in
            conn.respond(status: .ok, body: .string("[]"))
        }

        // Nested scope — both auth and rateLimit apply
        scope("/v2", through: [rateLimit]) {
            GET("/users") { conn in
                conn.respond(status: .ok, body: .string("v2"))
            }
        }
    }
}
// Matches: /health, /api/users, /api/v2/users

JSON & Form Parsing

Decode typed JSON with conn.decode(as:). Encode responses with conn.json(value:). Parse form bodies with conn.formParams. Dynamic JSON access with conn.jsonBody().

JSON.swift
POST("/users") { conn in
    let user = try conn.decode(as: CreateUser.self)
    let saved = try await db.insert(user)
    return try conn.json(status: .created, value: saved)
}

POST("/login") { conn in
    let username = conn.formParams["username"] ?? ""
    let password = conn.formParams["password"] ?? ""
    // authenticate...
    return conn.respond(status: .ok)
}

// Dynamic JSON — no struct needed
POST("/webhook") { conn in
    let body = try conn.jsonBody()
    let event = try body.string("event")
    return try conn.json(value: ["received": event])
}

Cookies

Read request cookies with conn.reqCookies. Set response cookies with putRespCookie using full RFC 6265 attributes. Delete cookies with deleteRespCookie.

Cookies.swift
GET("/dashboard") { conn in
    let sessionId = conn.reqCookies["session_id"]
    // look up session...
    return try conn.json(value: dashboard)
}

POST("/login") { conn in
    let cookie = Cookie(
        name: "session", value: token,
        path: "/", maxAge: 86400,
        secure: true, httpOnly: true,
        sameSite: .lax
    )
    return conn.putRespCookie(cookie)
        .respond(status: .ok, body: .string("OK"))
}

POST("/logout") { conn in
    conn.deleteRespCookie("session")
        .respond(status: .ok)
}

Streaming, SSE & Files

Stream responses with sendChunked. Format Server-Sent Events with sseEvent(). Serve files with sendFile — automatic MIME type detection and 64KB chunking.

Streaming.swift
// Server-Sent Events
GET("/events") { conn in
    conn.putRespContentType("text/event-stream")
        .sendChunked { writer in
            for i in 1...100 {
                writer.write(sseEvent(
                    data: "tick \(i)",
                    event: "heartbeat",
                    id: "\(i)"
                ))
                try await Task.sleep(for: .seconds(1))
            }
            writer.finish()
        }
}

// File serving with MIME inference
GET("/download/:file") { conn in
    let file = conn.params["file"] ?? ""
    return try conn.sendFile(path: "/var/www/\(file)")
}

Sessions & CSRF

Signed cookie-based sessions with HMAC-SHA256 via CryptoKit. CSRF protection validates tokens on state-changing requests. No external dependencies.

Sessions.swift
let secret = Data("my-32-byte-minimum-secret!!!".utf8)

let app = pipeline([
    sessionPlug(SessionConfig(secret: secret)),
    csrfProtection(),
    router.callAsFunction,
])

POST("/login") { conn in
    let user = try authenticate(conn)
    return conn
        .putSession(key: "user_id", value: user.id)
        .respond(status: .ok)
}

GET("/form") { conn in
    let (token, conn) = csrfToken(conn: conn)
    return conn.respond(body: .string("""
        <input name="_csrf_token" value="\(token)">
    """))
}

Service Injection

Inject shared services into the pipeline with assign(_:value:). Route handlers read from conn.assigns instead of capturing external references. Static values and per-request closures.

Injection.swift
let spectro = try SpectroClient(...)

let app = pipeline([
    requestId(),
    assign("spectro", value: spectro),   // static
    assign("trace") { UUID().uuidString }, // per-request
    router.callAsFunction,
])

GET("/donuts") { conn in
    let db = conn.assigns["spectro"] as! SpectroClient
    let donuts = try await db.repository().all(Donut.self)
    return try conn.json(value: donuts)
}

Typed Assigns

Define an AssignKey for compile-time-safe access. No string typos, no casts. Built-in keys for request ID, session, and remote IP with convenience accessors.

TypedAssigns.swift
// Define a typed key
enum SpectroKey: AssignKey {
    typealias Value = SpectroClient
}

// Write — typed
conn = conn.assign(SpectroKey.self, value: spectro)

// Read — no cast needed
let db = conn[SpectroKey.self]  // SpectroClient?

// Built-in convenience accessors
conn.requestId  // String?  (from RequestIdKey)
conn.session    // [String: String]?
conn.remoteIP   // String?

WebSocket

Define WebSocket routes with WS(). The upgrade handler inspects the HTTP connection (with assigns from upstream plugs). Message handlers are plain closures.

WebSocket.swift
let echo = WS("/ws/echo", onUpgrade: { conn in
    // Access auth, session, params from HTTP pipeline
    WSConnection(assigns: conn.assigns, send: { _ in })
}, onMessage: { ws, message in
    switch message {
    case .text(let text):
        try await ws.send(.text("Echo: \(text)"))
    case .binary(let data):
        try await ws.send(.binary(data))
    default: break
    }
})

// Hummingbird integration
let app = Application(
    responder: adapter,
    server: .nexusWebSocket(routes: [echo], plug: myPipeline)
)

Hummingbird Adapter

Plug your pipeline into Hummingbird with NexusHummingbirdAdapter. It converts requests to Connections and responses back. Errors become 500s per ADR-004.

Server.swift
import Nexus
import NexusRouter
import NexusHummingbird
import Hummingbird

let router = Router {
    GET("/hello") { conn in
        conn.respond(status: .ok, body: .string("Hello!"))
    }
}

let app = pipeline([logger, router])
let adapter = NexusHummingbirdAdapter(plug: app)

let server = Application(responder: adapter)
try await server.runService()
